    var audioCtx = new (window.AudioContext || window.webkitAudioContext)();

    function tocarSomPeca() {
        if (audioCtx.state === 'suspended') audioCtx.resume();
        var osc = audioCtx.createOscillator();
        var gain = audioCtx.createGain();
        osc.type = 'triangle';
        osc.frequency.setValueAtTime(120, audioCtx.currentTime); 
        osc.frequency.exponentialRampToValueAtTime(40, audioCtx.currentTime + 0.08);
        
        gain.gain.setValueAtTime(1.2, audioCtx.currentTime);
        gain.gain.exponentialRampToValueAtTime(0.01, audioCtx.currentTime + 0.08);
        
        osc.connect(gain);
        gain.connect(audioCtx.destination);
        osc.start();
        osc.stop(audioCtx.currentTime + 0.08);
    }

    function carregarPecas(piece) {
        return 'https://raw.githubusercontent.com/oakmac/chessboardjs/master/website/img/chesspieces/wikipedia/' + piece + '.png';
    }

    function formatarTempo(segundos) {
        var min = Math.floor(segundos / 60);
        var seg = segundos % 60;
        return (min < 10 ? '0' : '') + min + ':' + (seg < 10 ? '0' : '') + seg;
    }

    function aplicarCoresDosRelogios() {
        $('#clock-jogador').removeClass('clock-style-light clock-style-dark');
        $('#clock-ia').removeClass('clock-style-light clock-style-dark');

        if (corEscolhida === 'w') {
            $('#clock-jogador').addClass('clock-style-light');
            $('#clock-ia').addClass('clock-style-dark');
        } else {
            $('#clock-jogador').addClass('clock-style-dark');
            $('#clock-ia').addClass('clock-style-light');
        }
    }

    function atualizarDestaqueTurno() {
        $('.clock').removeClass('active-turn');
        if (game.game_over() || !jogoAtivo) return;
        if (game.turn() === corEscolhida) $('#clock-jogador').addClass('active-turn');
        else $('#clock-ia').addClass('active-turn');
    }

    function rodarCronometro() {
        if (!jogoAtivo || game.game_over()) return;
        if (game.turn() === corEscolhida) {
            tempoJogador--;
            $('#tempo-jogador').text(formatarTempo(tempoJogador));
            if (tempoJogador <= 0) {
                jogoAtivo = false;
                clearInterval(intervaloRelogio);
                alert("O seu tempo acabou! Vitória do JB CHESS por tempo.");
            }
        } else {
            tempoIA--;
            $('#tempo-ia').text(formatarTempo(tempoIA));
            if (tempoIA <= 0) {
                jogoAtivo = false;
                clearInterval(intervaloRelogio);
                alert("O tempo do JB CHESS acabou! Sua vitória por tempo.");
            }
        }
        atualizarDestaqueTurno();
    }

    function iniciarRelogio() {
        if (!intervaloRelogio) intervaloRelogio = setInterval(rodarCronometro, 1000);
        atualizarDestaqueTurno();
    }

    function atualizarHistoricoVisual() {
        var moves = game.history();
        var html = '';
        for (var i = 0; i < moves.length; i += 2) {
            var numLance = Math.floor(i / 2) + 1;
            html += '<span class="move-number">' + numLance + '. </span>';
            html += '<span class="move-text">' + moves[i] + '</span>';
            if (moves[i + 1]) html += '<span class="move-text">' + moves[i + 1] + '</span>';
            if (numLance % 2 === 0 && moves[i+1]) html += '<br>';
        }
        var $box = $('#history');
        $box.html(html);
        $box.scrollTop($box[0].scrollHeight); 
    }

    function onDragStart(source, piece, position, orientation) {
        if (!jogoAtivo || game.game_over()) return false;
        if (game.turn() === 'w' && piece.search(/^b/) !== -1) return false;
        if (game.turn() === 'b' && piece.search(/^w/) !== -1) return false;
        if (game.turn() !== corEscolhida) return false;
    }

    // CALIBRAGEM REFORÇADA PARA O NÍVEL FÁCIL (~1400 Lichess)
    function calcularMelhorLanceLocal() {
        var lancesPosiveis = game.moves({verbose: true});
        if (lancesPosiveis.length === 0) return null;

        var valoresPecas = { p: 10, n: 30, b: 30, r: 50, q: 90, k: 900 };

        var lancesAvaliados = lancesPosiveis.map(function(lance) {
            var pontuacao = 0;
            
            if (lance.captured) {
                pontuacao += valoresPecas[lance.captured] * 10;
            }
            
            if (lance.promotion) {
                pontuacao += 80;
            }

            // Centralização mais valorizada (melhora o jogo posicional)
            if (lance.piece === 'p' && (lance.to === 'd4' || lance.to === 'e4' || lance.to === 'd5' || lance.to === 'e5')) {
                pontuacao += 8;
            }
            if ((lance.piece === 'n' || lance.piece === 'b') && (lance.to.match(/[c-f][3-6]/))) {
                pontuacao += 6;
            }

            // Executa o lance temporariamente para ver se coloca o oponente em xeque
            game.move(lance);
            if (game.in_check()) pontuacao += 15;
            game.undo();

            return { lance: lance, score: pontuacao };
        });

        lancesAvaliados.sort(function(a, b) { return b.score - a.score; });

        // Redução da taxa de erro para 15% (deixando o computador mais astuto e estável)
        var indiceEscolhido = 0;
        if (Math.random() < 0.15 && lancesAvaliados.length > 1) {
            indiceEscolhido = Math.floor(Math.random() * Math.min(2, lancesAvaliados.length));
        }

        return lancesAvaliados[indiceEscolhido].lance;
    }

    function fazerJogadaDaIA() {
        if (!jogoAtivo || game.game_over()) return;
        var profundidade = $('#dificuldade').val(); 
        atualizarDestaqueTurno();

        if (profundidade === 'local') {
            window.setTimeout(function() {
                var lanceLocal = calcularMelhorLanceLocal();
                if (lanceLocal) {
                    game.move(lanceLocal);
                    board.position(game.fen());
                    tocarSomPeca();
                    atualizarHistoricoVisual();
                    verificarFimDeJogo();
                    atualizarDestaqueTurno();
                }
            }, 350); 
        } else {
            var fenAtual = game.fen();
            $.getJSON('https://stockfish.online/api/s/v2.php?fen=' + encodeURIComponent(fenAtual) + '&depth=' + profundidade, function(data) {
                if (data && data.success && data.bestmove) {
                    var melhorLance = data.bestmove.split(' ')[1]; 
                    var de = melhorLance.substring(0, 2);
                    var para = melhorLance.substring(2, 4);
                    var promocaoIA = melhorLance.length > 4 ? melhorLance.substring(4, 5) : 'q';

                    game.move({ from: de, to: para, promotion: promocaoIA });
                    board.position(game.fen());
                    tocarSomPeca();
                    atualizarHistoricoVisual();
                    verificarFimDeJogo();
                    atualizarDestaqueTurno();
                }
            });
        }
    }

    function processarMovimentoHumano(origem, destino) {
        var ehPromocao = false;
        var peao = game.get(origem);
        if (peao && peao.type === 'p') {
            if ((peao.color === 'w' && destino.search(/8/) !== -1) || (peao.color === 'b' && destino.search(/1/) !== -1)) {
                ehPromocao = true;
            }
        }
        var pecaEscolhida = 'q';
        if (ehPromocao) {
            var resposta = prompt("Promoção! Digite a letra da peça desejada:\nD = Dama\nT = Torre\nB = Bispo\nC = Cavalo", "D");
            if (resposta) {
                resposta = resposta.toUpperCase();
                if (resposta === 'D') pecaEscolhida = 'q';
                else if (resposta === 'T') pecaEscolhida = 'r';
                else if (resposta === 'B') pecaEscolhida = 'b';
                else if (resposta === 'C') pecaEscolhida = 'n';
            }
        }

        var move = game.move({ from: origem, to: destino, promotion: pecaEscolhida });
        if (move === null) return false;

        tocarSomPeca();
        atualizarHistoricoVisual();
        
        $('#seletor-tempo').prop('disabled', true);
        $('#cor-jogador').prop('disabled', true);
        $('#dificuldade').prop('disabled', true); 
        
        iniciarRelogio();
        window.setTimeout(fazerJogadaDaIA, 250);
        return true;
    }

    function onDrop(source, target) {
        removeHighlight();
        casaOrigemClicada = null;
        var sucesso = processarMovimentoHumano(source, target);
        if (!sucesso) return 'snapback';
    }

    function onSnapEnd() { board.position(game.fen()); }

    function removeHighlight() {
        $('#board .square-55d63').removeClass('highlight-square');
    }

    function addHighlight(square) {
        $('#board .square-' + square).addClass('highlight-square');
    }

    function tratarSelecaoDeCasa(square) {
        if (!jogoAtivo || game.game_over() || game.turn() !== corEscolhida) return;
        var pecaNaCasa = game.get(square);

        if (casaOrigemClicada === null) {
            if (pecaNaCasa && pecaNaCasa.color === corEscolhida) {
                casaOrigemClicada = square;
                removeHighlight();
                addHighlight(square);
            }
        } else {
            if (square === casaOrigemClicada) {
                casaOrigemClicada = null;
                removeHighlight();
            } else {
                var executou = processarMovimentoHumano(casaOrigemClicada, square);
                if (executou) {
                    board.position(game.fen());
                    casaOrigemClicada = null;
                    removeHighlight();
                } else {
                    if (pecaNaCasa && pecaNaCasa.color === corEscolhida) {
                        casaOrigemClicada = square;
                        removeHighlight();
                        addHighlight(square);
                    } else {
                        casaOrigemClicada = null;
                        removeHighlight();
                    }
                }
            }
        }
    }

    function verificarFimDeJogo() {
        if (game.in_checkmate()) {
            jogoAtivo = false;
            clearInterval(intervaloRelogio);
            alert("Fim de jogo! Xeque-mate.");

            clearInterval(intervaloRelogio);

        }
        atualizarDestaqueTurno();
    }

        $('#btn-empate').prop('disabled', false);

        $('#history').html('');
        
        aplicarCoresDosRelogios();
        game.reset();
        
        if (corEscolhida === 'b') board.orientation('black');
        else board.orientation('white');
        
        board.start();
        atualizarDestaqueTurno();
        
        if (corEscolhida === 'b') {
            iniciarRelogio();
            window.setTimeout(fazerJogadaDaIA, 500);
        }
    }

    function calcularVantagemMaterial() {
        var tabuleiro = game.board();
        var pontosBrancas = 0;
        var pontosPretas = 0;
        var valores = { 'p': 1, 'n': 3, 'b': 3, 'r': 5, 'q': 9, 'k': 0 };

        for (var r = 0; r < 8; r++) {
            for (var c = 0; c < 8; c++) {
                var casa = tabuleiro[r][c];
                if (casa) {
                    if (casa.color === 'w') pontosBrancas += valores[casa.type];
                    else pontosPretas += valores[casa.type];
                }
            }
        }
        var pontosIA = (corEscolhida === 'b') ? pontosBrancas : pontosPretas;
        var pontosJogador = (corEscolhida === 'b') ? pontosPretas : pontosBrancas;
        return pontosIA - pontosJogador; 
    }

        if (confirm("Tem certeza que deseja desistir da partida?")) {

            clearInterval(intervaloRelogio);
            $('#btn-empate').prop('disabled', true);
            $('#btn-desistir').prop('disabled', true);
            atualizarDestaqueTurno();
            alert("Você abandonou. Vitória do JB CHESS!");
        }
    });

    $('#btn-empate').on('click', function() {
        if (!jogoAtivo || game.game_over()) return;
        
        var vantagemIA = calcularVantagemMaterial();
        var profundidade = $('#dificuldade').val();
        var aceitaEmpate = false;

        if (vantagemIA > 1) {
            alert("O JB CHESS recusou a sua proposta de empate. A IA avalia que tem vantagem material suficiente para jogar pela vitória!");
            return;
        } else if (vantagemIA < -1) {
            aceitaEmpate = true;
        } else {
            var numeroAleatorio = Math.random();
            if (profundidade === 'local') { if (numeroAleatorio < 0.70) aceitaEmpate = true; } 
            else if (profundidade === '6') { if (numeroAleatorio < 0.40) aceitaEmpate = true; } 
            else if (profundidade === '10') { if (numeroAleatorio < 0.10) aceitaEmpate = true; }
            else if (profundidade === '14') { if (numeroAleatorio < 0.02) aceitaEmpate = true; }
        }

        if (aceitaEmpate) {
            jogoAtivo = false;
            clearInterval(intervaloRelogio);
